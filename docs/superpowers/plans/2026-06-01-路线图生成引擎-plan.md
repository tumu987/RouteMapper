**Goal:** 构建可复用的自驾路线图生成引擎，从 JSON 配置文件生成高分辨率路线图 PNG。

**Architecture:** 三层分离 — config.py 负责数据加载/验证，layout.py 负责碰撞检测与坐标计算，renderer.py 负责 matplotlib 绘制，generate.py 负责编排整个流程。

**Tech Stack:** Python 3.9+, matplotlib, cartopy, numpy, shapely

**参考脚本:** nanjiang_v3.py（矩形碰撞引擎）, generate_final.py（行程表+景点）, generate_route_v5.py（省界颜色）

```python
def load_config(path: str) -> Dict[str, Any]:
    """加载 JSON 配置文件，补充默认值，验证必要字段。"""
    with open(path, "r", encoding="utf-8") as f:
        cfg = json.load(f)

    # 验证必要字段
    _validate(cfg)

    # 补充默认值
    cfg.setdefault("output", {})
    for k, v in DEFAULT_OUTPUT.items():
        cfg["output"].setdefault(k, v)

    # 自动生成 day_colors（如果未提供）
    if "day_colors" not in cfg:
        max_day = _max_day(cfg)
        cfg["day_colors"] = {
            str(d): AUTO_DAY_COLORS[(d - 1) % len(AUTO_DAY_COLORS)]
            for d in range(1, max_day + 1)
        }

    # 自动生成 map_extent（如果未提供）
    if "map_extent" not in cfg:
        cfg["map_extent"] = _auto_extent(cfg)

    return cfg


def _max_day(cfg: dict) -> int:
    """计算最大天数。"""
    days = set()
    for seg in cfg.get("segments", []):
        days.add(seg["day"])
    for d in cfg.get("rest_days", {}):
        days.add(int(d))
    for d in cfg.get("day_attractions", {}):
        days.add(int(d))
    return max(days) if days else 1


def _auto_extent(cfg: dict) -> List[float]:
    """根据城市坐标自动计算地图范围。"""
    cities = cfg["cities"]
    lons = [c["lon"] for c in cities]
    lats = [c["lat"] for c in cities]
    margin = 1.0
    return [
        min(lons) - margin,
        max(lons) + margin,
        min(lats) - margin,
        max(lats) + margin,
    ]


def _validate(cfg: dict) -> None:
    """验证配置必要字段，不合法时抛出 ValueError。"""
    if "cities" not in cfg or not cfg["cities"]:
        raise ValueError("配置缺少 cities 字段")
    if "segments" not in cfg:
        raise ValueError("配置缺少 segments 字段")
    if "title" not in cfg:
        raise ValueError("配置缺少 title 字段")

    # 验证城市索引引用
    n = len(cfg["cities"])
    for i, seg in enumerate(cfg["segments"]):
        for key in ("from_index", "to_index"):
            idx = seg.get(key)
            if idx is None or idx < 0 or idx >= n:
                raise ValueError(f"segments[{i}].{key}={idx} 越界（城市数={n}）")

    # 验证 rest_days 中的城市名存在
    city_names = {c["name"] for c in cfg["cities"]}
    for day, name in cfg.get("rest_days", {}).items():
        if name not in city_names:
            raise ValueError(f"rest_days day {day}: 城市 '{name}' 不在 cities 列表中")


def get_city_index(cfg: dict, name: str) -> Optional[int]:
    """根据城市名查找索引。"""
    for c in cfg["cities"]:
        if c["name"] == name:
            return c["index"]
    return None
```

```python
"""布局引擎：碰撞检测 + 逐层放置 + 密度感知。"""
import math
from typing import List, Tuple, Optional


class LayoutEngine:
    """管理元素放置位置和碰撞检测。

    所有坐标使用地图坐标（经度、纬度）。
    placed 列表存储已放置元素的矩形包围盒：[(cx, cy, hw, hh, label), ...]
    route_segs 列表存储路线段用于碰撞检测：[(x1, y1, x2, y2, half_width), ...]
    """

    def __init__(self):
        self.placed: List[Tuple[float, float, float, float, str]] = []
        self.route_segs: List[Tuple[float, float, float, float, float]] = []

    # ── 文字尺寸估计 ──

    @staticmethod
    def text_half_size(text: str, pt: float) -> Tuple[float, float]:
        """估算文字标签在地图坐标中的半宽、半高。

        校准基准：200px/°，华文黑体 Medium。
        """
        hw = len(text) * pt * 0.0031 + 0.005
        hh = pt * 0.0036 + 0.005
        return hw, hh

    # ── 碰撞检测 ──

    @staticmethod
    def rect_overlap(cx1: float, cy1: float, hw1: float, hh1: float,
                     cx2: float, cy2: float, hw2: float, hh2: float,
                     margin: float = 0.0) -> bool:
        """两个轴对齐矩形是否重叠。"""
        return (abs(cx1 - cx2) < hw1 + hw2 + margin and
                abs(cy1 - cy2) < hh1 + hh2 + margin)

    @staticmethod
    def dist_to_segment(x: float, y: float,
                        x1: float, y1: float,
                        x2: float, y2: float) -> float:
        """点到线段的最短距离。"""
        dx, dy = x2 - x1, y2 - y1
        if abs(dx) < 1e-8 and abs(dy) < 1e-8:
            return math.hypot(x - x1, y - y1)
        t = ((x - x1) * dx + (y - y1) * dy) / (dx * dx + dy * dy)
        if t <= 0:
            return math.hypot(x - x1, y - y1)
        if t >= 1:
            return math.hypot(x - x2, y - y2)
        px = x1 + t * dx
        py = y1 + t * dy
        return math.hypot(x - px, y - py)

    def route_collision(self, cx: float, cy: float,
                        hw: float, hh: float, margin: float = 0.06) -> bool:
        """检查矩形是否与任何路线段碰撞。

        返回 True = 碰撞，False = 安全。
        """
        for x1, y1, x2, y2, hlw in self.route_segs:
            d = self.dist_to_segment(cx, cy, x1, y1, x2, y2)
            dx, dy = x2 - x1, y2 - y1
            length = math.hypot(dx, dy)
            if length < 1e-8:
                continue
            nx, ny = -dy / length, dx / length
            r_eff = abs(nx) * hw + abs(ny) * hh
            if d <= r_eff + hlw + margin:
                return True
        return False

    # ── 放置管理 ──

    def check(self, cx: float, cy: float, hw: float, hh: float,
              margin: float = 0.04,
              route_margin: float = 0.06) -> bool:
        """检查位置是否可用（对所有已放置元素 + 路线段）。"""
        for pcx, pcy, phw, phh, _ in self.placed:
            if self.rect_overlap(cx, cy, hw, hh, pcx, pcy, phw, phh, margin):
                return False
        if self.route_collision(cx, cy, hw, hh, route_margin):
            return False
        return True

    def place(self, cx: float, cy: float, hw: float, hh: float,
              label: str = "") -> None:
        """注册已放置元素。"""
        self.placed.append((cx, cy, hw, hh, label))

    def register_route(self, x1: float, y1: float,
                       x2: float, y2: float,
                       half_width: float = 0.03) -> None:
        """注册路线段到碰撞系统。"""
        self.route_segs.append((x1, y1, x2, y2, half_width))

    # ── 辅助几何 ──

    @staticmethod
    def normal_offset(x1: float, y1: float, x2: float, y2: float,
                      dist: float = 0.15, side: float = 1.0) -> Tuple[float, float]:
        """返回垂直于路线的偏移量。side=1 右侧, side=-1 左侧。"""
        dx, dy = x2 - x1, y2 - y1
        length = math.hypot(dx, dy)
        if length < 0.001:
            return 0.0, 0.0
        nx, ny = -dy / length * side, dx / length * side
        return nx * dist, ny * dist

    def density_side(self, mx: float, my: float,
                     nx: float, ny: float,
                     radius: float = 0.5) -> int:
        """密度感知选侧：统计两侧已放元素数量，返回稀疏侧（1 或 -1）。"""
        scores = []
        for side in (1, -1):
            sx = mx + nx * 0.3 * side
            sy = my + ny * 0.3 * side
            count = sum(
                1 for px, py, _, _, _ in self.placed
                if math.hypot(sx - px, sy - py) < radius
            )
            scores.append((count, side))
        scores.sort()
        return scores[0][1]  # 返回元素少的侧

    def search_position(self, mx: float, my: float,
                        nx: float, ny: float,
                        hw: float, hh: float,
                        preferred_side: int,
                        base_offset: float = 0.12,
                        margin: float = 0.04,
                        route_margin: float = 0.06) -> Optional[Tuple[float, float, int]]:
        """在路线一侧递进搜索可用位置。

        返回 (x, y, side) 或 None。
        """
        for mult in (1.0, 1.5, 2.0, 2.5, 3.0, 3.5, 4.0):
            for side in (preferred_side, -preferred_side):
                ox = nx * base_offset * mult * side
                oy = ny * base_offset * mult * side
                tx, ty = mx + ox, my + oy
                if self.check(tx, ty, hw, hh, margin, route_margin):
                    return tx, ty, side
        return None

    def search_8dir(self, mx: float, my: float,
                    hw: float, hh: float,
                    margin: float = 0.04) -> Optional[Tuple[float, float]]:
        """8 方向 fallback 搜索。"""
        for angle_deg in (45, 135, 225, 315, 0, 90, 180, 270):
            rad = math.radians(angle_deg)
            for dist in (0.25, 0.35, 0.50):
                tx = mx + dist * math.cos(rad)
                ty = my + dist * math.sin(rad)
                if self.check(tx, ty, hw, hh, margin):
                    return tx, ty
        return None
```

```python
    # ── 天次放置 ──

    def place_travel_day(self, text: str, seg_indices: List[int],
                         all_segments: list, cities: list) -> Tuple[float, float, str]:
        """放置行车段天次标注。

        Args:
            text: 如 "D1"
            seg_indices: 该天包含的路段索引列表
            all_segments: 全部路段数据
            cities: 全部城市数据

        Returns:
            (x, y, ha) — 标注位置和水平对齐方式
        """
        # 计算多段中点
        segs = [all_segments[si] for si in seg_indices]
        if len(segs) == 1:
            s = segs[0]
            x1, y1 = cities[s["from_index"]]["lon"], cities[s["from_index"]]["lat"]
            x2, y2 = cities[s["to_index"]]["lon"], cities[s["to_index"]]["lat"]
            mx, my = (x1 + x2) / 2, (y1 + y2) / 2
        else:
            total = 0.0
            seg_data = []
            for s in segs:
                x1, y1 = cities[s["from_index"]]["lon"], cities[s["from_index"]]["lat"]
                x2, y2 = cities[s["to_index"]]["lon"], cities[s["to_index"]]["lat"]
                d = math.hypot(x2 - x1, y2 - y1)
                seg_data.append(((x1, y1), (x2, y2), d))
                total += d
            target = total / 2
            cum = 0.0
            mx = my = 0.0
            for (x1, y1), (x2, y2), d in seg_data:
                prev = cum
                cum += d
                if cum >= target:
                    frac = (target - prev) / d if d > 0 else 0
                    mx = x1 + (x2 - x1) * frac
                    my = y1 + (y2 - y1) * frac
                    break

        # 法线方向
        last = segs[-1]
        x1, y1 = cities[last["from_index"]]["lon"], cities[last["from_index"]]["lat"]
        x2, y2 = cities[last["to_index"]]["lon"], cities[last["to_index"]]["lat"]
        dx, dy = x2 - x1, y2 - y1
        length = math.hypot(dx, dy)
        if length < 1e-6:
            nx, ny = 0.0, 1.0
        else:
            nx, ny = -dy / length, dx / length

        # 最小偏移（保证文字外缘不压路线）
        hw, hh = self.text_half_size(text, 18)
        r_eff = abs(nx) * hw + abs(ny) * hh
        base_off = max(0.12, r_eff + 0.03 + 0.06)

        # 密度感知选侧，优先去距离/时间对面
        preferred = self.density_side(mx, my, nx, ny)

        # 递进搜索
        result = self.search_position(mx, my, nx, ny, hw, hh,
                                      preferred, base_off, 0.04, 0.06)
        if result:
            tx, ty, side = result
            ha = "left" if side > 0 else "right"
            self.place(tx, ty, hw, hh, f"day_{text}")
            return tx, ty, ha

        # Fallback: 8 方向
        fb = self.search_8dir(mx, my, hw, hh)
        if fb:
            tx, ty = fb
            ha = "left" if tx >= mx else "right"
            self.place(tx, ty, hw, hh, f"day_{text}")
            return tx, ty, ha

        # 最终 fallback
        tx, ty = mx + 0.3, my + 0.2
        self.place(tx, ty, hw, hh, f"day_{text}")
        return tx, ty, "left"

    def place_rest_day(self, day_num: str, city_name: str,
                       cities: list, crs_cos: float) -> Tuple[float, float, str, float, str]:
        """放置休息日标注（城市正下方）。

        Returns:
            (d_x, d_y, d_ha, c_x, c_y) — 天次位置和城市全称位置
        """
        ci = next(i for i, c in enumerate(cities) if c["name"] == city_name)
        clon, clat = cities[ci]["lon"], cities[ci]["lat"]
        cr = _city_radius(city_name)
        city_full = f"{city_name}全天"

        hw_d, hh_d = self.text_half_size(day_num, 16)
        hw_c, hh_c = self.text_half_size(city_full, 14)
        LINE = 0.10

        def pair_fits(bx: float, by: float) -> bool:
            for px, py, phw, phh, pl in self.placed:
                if pl.startswith(f"city_{city_name}"):
                    continue
                if self.rect_overlap(bx, by, hw_d, hh_d, px, py, phw, phh, 0.03):
                    return False
                if self.rect_overlap(bx, by - LINE, hw_c, hh_c, px, py, phw, phh, 0.03):
                    return False
            if self.route_collision(bx, by, hw_d, hh_d):
                return False
            if self.route_collision(bx, by - LINE, hw_c, hh_c):
                return False
            # 不压其他城市
            for ci2, c2 in enumerate(cities):
                if ci2 == ci:
                    continue
                cr2 = _city_radius(c2["name"])
                if abs(bx - c2["lon"]) < hw_c + cr2 + 0.15 and \
                   abs(by - LINE - c2["lat"]) < hh_c + cr2 * crs_cos + 0.15:
                    return False
            return True

        base_dy = cr + 0.08
        for extra in (0, 0.03, 0.05, 0.08, 0.10, 0.15, 0.20, 0.25, 0.30, 0.35, 0.40):
            dy = base_dy + extra
            if pair_fits(clon, clat - dy):
                self.place(clon, clat - dy, hw_d, hh_d, f"day_{day_num}")
                self.place(clon, clat - dy - LINE, hw_c, hh_c, f"day_{city_full}")
                return clon, clat - dy, "center", clon, clat - dy - LINE

        # Fallback: 8 方向
        for angle_deg in (0, 180, 90, 270, 45, 135, 225, 315):
            rad = math.radians(angle_deg)
            tx = clon + 0.5 * math.cos(rad)
            ty = clat + 0.5 * math.sin(rad)
            if pair_fits(tx, ty):
                ha = "center" if abs(math.cos(rad)) < 0.1 else ("left" if math.cos(rad) > 0 else "right")
                self.place(tx, ty, hw_d, hh_d, f"day_{day_num}")
                self.place(tx, ty - LINE, hw_c, hh_c, f"day_{city_full}")
                return tx, ty, ha, tx, ty - LINE

        # 最终 fallback
        tx, ty = clon, clat - 0.55
        self.place(tx, ty, hw_d, hh_d, f"day_{day_num}")
        self.place(tx, ty - LINE, hw_c, hh_c, f"day_{city_full}")
        return tx, ty, "center", tx, ty - LINE

    # ── 距离/时间放置 ──

    def place_dist_time(self, dist_text: str, time_text: str,
                        seg_index: int, all_segments: list,
                        cities: list) -> Tuple[float, float, float, float]:
        """放置距离和时间标注。

        距离在上，时间在下，纵向排列。主动避让天次标注。

        Returns:
            (dist_x, dist_y, time_x, time_y)
        """
        seg = all_segments[seg_index]
        x1, y1 = cities[seg["from_index"]]["lon"], cities[seg["from_index"]]["lat"]
        x2, y2 = cities[seg["to_index"]]["lon"], cities[seg["to_index"]]["lat"]
        mx, my = (x1 + x2) / 2, (y1 + y2) / 2

        dx, dy = x2 - x1, y2 - y1
        length = math.hypot(dx, dy)
        nx, ny = -dy / length, dx / length

        hw_d, hh_d = self.text_half_size(dist_text, 16)
        hw_t, hh_t = self.text_half_size(time_text, 14)
        LINE = 0.10

        # 最小偏移
        r_eff = max(abs(nx) * hw_d + abs(ny) * hh_d,
                    abs(nx) * hw_t + abs(ny) * hh_t)
        base_off = max(0.12, r_eff + 0.03 + 0.06)

        # 天次在哪侧？
        day_num = seg["day"]
        day_side = None
        for px, py, _, _, pl in self.placed:
            if pl == f"day_D{day_num}" or pl.startswith(f"day_D{day_num}（"):
                day_side = 1 if (px - mx) * nx + (py - my) * ny >= 0 else -1
                break

        # 优先选天次对面
        preferred = -day_side if day_side else self.density_side(mx, my, nx, ny)

        # 搜索（距离+时间作为整体）
        def pair_fits(bx: float, by: float) -> bool:
            if not self.check(bx, by, hw_d, hh_d, 0.04, 0.06):
                return False
            if not self.check(bx, by - LINE, hw_t, hh_t, 0.04, 0.06):
                return False
            # 时间不压城市
            for c in cities:
                cr = _city_radius(c["name"])
                if abs(bx - c["lon"]) < hw_t + cr + 0.15 and \
                   abs(by - LINE - c["lat"]) < hh_t + cr + 0.15:
                    return False
            return True

        # 递进搜索
        for mult in (1.0, 1.5, 2.0, 2.5, 3.0, 3.5, 4.0, 5.0):
            for side in (preferred, -preferred):
                ox = nx * base_off * mult * side
                oy = ny * base_off * mult * side
                bx, by = mx + ox, my + oy
                if pair_fits(bx - 0.02, by):
                    self.place(bx - 0.02, by, hw_d, hh_d, f"dist_{seg_index}")
                    self.place(bx - 0.02, by - LINE, hw_t, hh_t, f"time_{seg_index}")
                    return bx - 0.02, by, bx - 0.02, by - LINE

        # 允许分开距离和时间
        for mult in (1.0, 1.5, 2.0, 2.5, 3.0):
            for side in (preferred, -preferred):
                ox = nx * base_off * mult * side
                oy = ny * base_off * mult * side
                bx, by = mx + ox, my + oy
                if self.check(bx, by, hw_d, hh_d):
                    self.place(bx, by, hw_d, hh_d, f"dist_{seg_index}")
                    # 时间在附近搜索
                    for tdx, tdy in ((0, -LINE), (0.15, -LINE), (-0.15, -LINE)):
                        tx, ty = bx + tdx, by + tdy
                        if self.check(tx, ty, hw_t, hh_t):
                            self.place(tx, ty, hw_t, hh_t, f"time_{seg_index}")
                            return bx, by, tx, ty
                    # 如果时间放不下，保持原地
                    tx, ty = bx, by - LINE
                    self.place(tx, ty, hw_t, hh_t, f"time_{seg_index}")
                    return bx, by, tx, ty

        # Fallback
        tx, ty = mx + 0.4, my
        self.place(tx, ty, hw_d, hh_d, f"dist_{seg_index}")
        self.place(tx, ty - LINE, hw_t, hh_t, f"time_{seg_index}")
        return tx, ty, tx, ty - LINE


def _city_radius(name: str) -> float:
    """城市圈半径（地图坐标）。"""
    if len(name) <= 2:
        return 0.13
    if len(name) == 3:
        return 0.14
    return 0.15
```

```python
    # ── 行程表放置 ──

    def place_itinerary(self, lines: List[str],
                        map_extent: List[float],
                        cities: list,
                        font_size: float = 14,
                        line_height: float = 0.21,
                        char_width: float = 0.09) -> Optional[Tuple[float, float, float, float]]:
        """网格搜索最佳空白区域放置行程表。

        Returns:
            (cx, cy, tw, th) — 左上角坐标和宽高，或 None
        """
        tw = min(max(len(l) for l in lines) * char_width + 0.3, 3.8)
        th = len(lines) * line_height + 0.10
        west, east, south, north = map_extent

        cx_min = west + 0.2
        cx_max = east - 0.2 - tw
        cy_max = north - 0.9 - 0.15 - th / 2  # 标题下方
        cy_min = south + 0.2 + th / 2

        if cx_min > cx_max or cy_min > cy_max:
            return None

        best, best_score = None, float("inf")
        y = cy_max
        while y >= cy_min:
            x = cx_min
            while x <= cx_max:
                bx, by = x + tw / 2, y
                # 评分：附近元素数 + 距顶边惩罚
                count = sum(1 for px, py, _, _, _ in self.placed
                           if math.hypot(bx - px, by - py) < 1.0)
                count += sum(3 for c in cities
                            if math.hypot(c["lon"] - bx, c["lat"] - by) < 1.0)
                score = count + (north - y) * 0.02

                if score < best_score:
                    hw, hh = tw / 2, th / 2
                    ok = True
                    for px, py, phw, phh, _ in self.placed:
                        if abs(bx - px) < hw + phw + 0.08 and \
                           abs(by - py) < hh + phh + 0.08:
                            ok = False
                            break
                    if ok:
                        best = (x, y, tw, th)
                        best_score = score
                x += 0.15
            y -= 0.15

        if best:
            x, y, tw, th = best
            self.place(x + tw / 2, y, tw / 2, th / 2, "itinerary")
            return best
        return None

    def place_itinerary_two_col(self, lines: List[str],
                                 map_extent: List[float],
                                 cities: list) -> Optional[List[Tuple[float, float, float, float]]]:
        """双列行程表放置（单列放不下时的 fallback）。

        Returns:
            [(cx1, cy1, tw1, th1), (cx2, cy2, tw2, th2)] 或 None
        """
        mid = len(lines) // 2
        col1, col2 = lines[:mid], lines[mid:]

        char_width = 0.09
        line_height = 0.21
        tw1 = min(max(len(l) for l in col1) * char_width + 0.3, 3.8)
        tw2 = min(max(len(l) for l in col2) * char_width + 0.3, 3.8)
        th1 = len(col1) * line_height + 0.10

        west, east, south, north = map_extent
        cy_max = north - 0.9 - 0.15 - th1 / 2
        cy_min = south + 0.2 + th1 / 2

        best = None
        best_score = float("inf")
        y = cy_max
        while y >= cy_min:
            x = west + 0.2
            while x + tw1 + 0.2 + tw2 <= east - 0.2:
                bx, by = x + tw1 / 2, y
                hw, hh = tw1 / 2, th1 / 2
                ok = all(
                    abs(bx - px) >= hw + phw + 0.08 or
                    abs(by - py) >= hh + phh + 0.08
                    for px, py, phw, phh, _ in self.placed
                )
                if ok:
                    # 检查第二列
                    x2 = x + tw1 + 0.2
                    bx2, by2 = x2 + tw2 / 2, y
                    hw2, hh2 = tw2 / 2, th1 / 2
                    ok2 = all(
                        abs(bx2 - px) >= hw2 + phw + 0.08 or
                        abs(by2 - py) >= hh2 + phh + 0.08

            y -= 0.15

            for x, y, tw, th in best:
                self.place(x + tw / 2, y, tw / 2, th / 2, "itinerary")
        return best
```

```python
    ax.text(tx, ty, time_text, fontproperties=F(14), color="#E67E22",
            ha="center", va="center", zorder=30, path_effects=ST(4))


# ── 景点 ──

def render_attraction_outside(ax, name: str, alat: float, alon: float,
                               is_primary: bool, x: float, y: float,
                               ha: str, leader_info: dict) -> None:
    """渲染城圈外景点：打点 + 紧邻标签 + 可选引线。"""
    color = "#8B4513" if is_primary else "#27AE60"
    sz = 14 if is_primary else 13
    mfc = color if is_primary else "none"
    ls = "-" if is_primary else "--"

    # 打点
    ax.plot(alon, alat, "o", color=color, markersize=5,
            markerfacecolor=mfc, markeredgewidth=2,
            transform=ccrs.PlateCarree(), zorder=16)

    # 引线（如果文字距点超过 0.20°）
    if leader_info.get("need_leader", False):
        lx, ly = leader_info.get("lx", x), leader_info.get("ly", y)
        ax.plot([lx, alon], [ly, alat], color=color,
                linewidth=1.5, linestyle=ls, alpha=0.7,
                transform=ccrs.PlateCarree(), zorder=15)

    # 标签
    ax.text(x, y, name, fontproperties=F(sz), color=color,
            ha=ha, va="center", zorder=17, path_effects=ST(3.5))


def render_grouped_attractions(ax, city_name: str, city_lat: float,
                                city_lon: float, anchor_x: float,
                                anchor_y: float, side: int,
                                prim_str: str, sec_str: str) -> None:
    """渲染城圈内景点合并引线。"""
    cr = _city_radius(city_name)
    gap = 0.10 if len(city_name) >= 3 else 0.03

    # 引线起点（城圈边缘）
    angle = math.atan2(anchor_y - city_lat, anchor_x - city_lon)
    if abs(math.cos(angle)) < 0.01:
        ed_x, ed_y = city_lon, city_lat + math.copysign(cr, math.sin(angle))
    else:
        ed_x = city_lon + cr * math.cos(angle)
        ed_y = city_lat + cr * math.sin(angle) * math.cos(
            math.atan2(anchor_y - city_lat, anchor_x - city_lon)
        )
    # 简化：用 polar
    ed_x = city_lon + cr * math.cos(angle)
    ed_y = city_lat + cr * math.sin(angle)

    ax.plot([anchor_x, ed_x], [anchor_y, ed_y], color="#8B4513",
            linewidth=1.5, alpha=0.7,
            transform=ccrs.PlateCarree(), zorder=15)

    ha = "left" if side > 0 else "right"
    LINE = 0.10
    if prim_str:
        ax.text(anchor_x, anchor_y, prim_str, fontproperties=F(14),
                color="#8B4513", ha=ha, va="center", zorder=17,
                path_effects=ST(3.5))
    if sec_str:
        ax.text(anchor_x, anchor_y - LINE, sec_str, fontproperties=F(13),
                color="#27AE60", ha=ha, va="center", zorder=17,
                path_effects=ST(3.5))


# ── 标题 ──

def render_title(ax, cfg: dict, center_x: float) -> None:
    """渲染主标题和副标题。"""
    cities = cfg["cities"]
    extent = cfg["map_extent"]
    title = cfg["title"]
    subtitle = cfg.get("subtitle", "")
    days = _count_total_days(cfg)
    total_km = sum(int(s["distance"].rstrip("km")) for s in cfg["segments"])

    title_y = extent[3] - 0.6
    sub_y = extent[3] - 0.9

    ax.text(center_x, title_y,
            f"{cities[0]['name']}→{cities[-1]['name']} · {title}",
            fontproperties=F(30), color="#2C3E50",
            ha="center", va="bottom", zorder=100, path_effects=ST(6))

    sub_text = subtitle
    if days:
        sub_text += f"  ·  {days}天"
    if total_km:
        sub_text += f" · 约{total_km}km"

    ax.text(center_x, sub_y, sub_text,
            fontproperties=F(18), color="#7F8C8D",
            ha="center", va="bottom", zorder=100, path_effects=ST(4))


# ── 行程表 ──

def render_itinerary(ax, lines: List[str], cx: float, cy: float,
                     tw: float, th: float, font_size: float = 14,
                     line_height: float = 0.21) -> None:
    """渲染行程表。"""
    for i, line in enumerate(lines):
        yy = cy + th / 2 - 0.08 - i * line_height
        ax.text(cx, yy, line, fontproperties=F(font_size),
                color="#2C3E50", ha="left", va="top",
                zorder=95, path_effects=ST(1.5))


# ── 辅助函数 ──

def _city_radius(name: str) -> float:
    if len(name) <= 2:
        return 0.13
    if len(name) == 3:
        return 0.14
    return 0.15


def _count_total_days(cfg: dict) -> int:
    """计算总天数。"""
    days = set()
    for seg in cfg.get("segments", []):
        days.add(seg["day"])
    for d in cfg.get("rest_days", {}):
        days.add(int(d))
    for d in cfg.get("day_attractions", {}):
        days.add(int(d))
    for c in cfg.get("cities", []):
        for a in c.get("attractions_primary", []):
            if "day" in a:
                days.add(a["day"])
        for a in c.get("attractions_secondary", []):
            if "day" in a:
                days.add(a["day"])
    return max(days) if days else 0
```

```python
#!/usr/bin/env python3
"""路线图生成主入口。

用法:
    python generate.py routes/nanjiang.json
    python generate.py routes/nanjiang.json --output ~/Desktop/my_map.png
"""
import argparse
import math
import os
import shutil
import sys
import warnings

warnings.filterwarnings("ignore")
import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import cartopy.crs as ccrs

from config import load_config, EN_PROV_MAP, PROV_CN_NAMES, PROVINCE_COLORS
from layout import LayoutEngine
from renderer import (
    F, ST, render_base_map, render_provinces, render_all_routes,
    render_city_node, render_day_label, render_rest_day_label,
    render_dist_time, render_attraction_outside, render_grouped_attractions,
    render_title, render_itinerary,
)


def generate(cfg: dict) -> str:
    """根据配置生成路线图，返回输出文件路径。"""
    output = cfg["output"]
    cities = cfg["cities"]
    segments = cfg["segments"]
    extent = list(cfg["map_extent"])
    day_colors = cfg["day_colors"]
    rest_days = cfg.get("rest_days", {})
    day_attractions = cfg.get("day_attractions", {})

    # ── 计算纬度补偿 ──
    avg_lat = sum(c["lat"] for c in cities) / len(cities)
    crs_cos = math.cos(math.radians(avg_lat))

    # ── 预计算天数 ──
    day_to_segs = {}
    for si, seg in enumerate(segments):
        d = seg["day"]
        day_to_segs.setdefault(d, []).append(si)

    all_seg_days = set(day_to_segs.keys())
    all_rest_days = set(int(k) for k in rest_days.keys())
    all_attr_days = set(int(k) for k in day_attractions.keys())
    max_day = max(all_seg_days | all_rest_days | all_attr_days | {1})
    all_days = sorted(range(1, max_day + 1))

    # ── 自动扩展北边界（容纳行程表） ──
    if output.get("auto_extent", True):
        LINE_H = 0.21
        table_lines = _build_itinerary_lines(cfg, all_days, day_to_segs)
        th_table = len(table_lines) * LINE_H + 0.10
        peak_node_top = max(
            c["lat"] + _city_radius(c["name"]) + 0.04 for c in cities
        )
        table_north = peak_node_top + 0.03 + 1.05 + th_table
        if table_north > extent[3]:
            print(f"  自动扩展北边界: {extent[3]:.1f}° → {table_north:.1f}°")
            extent[3] = table_north

    # ── 创建图形 ──
    fig = plt.figure(
        figsize=(output["width_inch"], output["height_inch"]),
        facecolor="white",
    )
    ax = fig.add_subplot(111, projection=ccrs.PlateCarree())
    fig.subplots_adjust(left=0.01, right=0.99, top=0.99, bottom=0.01)

    ax.set_extent(extent)
    ax.set_aspect(1.0 / crs_cos, adjustable="box")

    # ── L1 底图 ──
    print("底图...")
    render_base_map(ax, extent)
    render_provinces(ax, extent, PROVINCE_COLORS, EN_PROV_MAP, PROV_CN_NAMES)

    # ── L2 路线 ──
    print("路线...")
    route_pairs = render_all_routes(ax, segments, cities, day_colors)

    # 布局引擎初始化
    layout = LayoutEngine()

    # 注册路线到碰撞系统
    for fi, ti in route_pairs:
        x1, y1 = cities[fi]["lon"], cities[fi]["lat"]
        x2, y2 = cities[ti]["lon"], cities[ti]["lat"]
        layout.register_route(x1, y1, x2, y2, 0.03)

    # ── L3 城市节点 ──
    print("城市...")
    for c in cities:
        render_city_node(ax, c["name"], c["lon"], c["lat"],
                         c["color"], crs_cos)
        r = _city_radius(c["name"])
        layout.place(c["lon"], c["lat"], r + 0.04, r * crs_cos + 0.04,
                     f"city_{c['name']}")

    # ── L4 天次标注 ──
    print("天次...")
    for day in all_days:
        seg_indices = day_to_segs.get(day, [])
        if str(day) in rest_days:
            rest_city = rest_days[str(day)]
            dx, dy, dha, cx, cy = layout.place_rest_day(
                f"D{day}", rest_city, cities, crs_cos
            )
            render_rest_day_label(
                ax, f"D{day}", f"{rest_city}全天",
                dx, dy, dha, cx, cy,
            )
        elif seg_indices:
            tx, ty, ha = layout.place_travel_day(
                f"D{day}", seg_indices, segments, cities,
            )
            render_day_label(ax, f"D{day}", tx, ty, ha)
        # else: 驻留日无路段，不在地图上标天次

    # ── L5 距离/时间 ──
    print("距离/时间...")
    # 去重往返段（只标一个方向的 dist/time）
    round_trip_skip = set()
    for i, s1 in enumerate(segments):
        for j, s2 in enumerate(segments):
            if i != j and s1["from_index"] == s2["to_index"] and \
               s1["to_index"] == s2["from_index"]:
                if s1["from_index"] > s1["to_index"]:
                    round_trip_skip.add(i)
                break

    for si, seg in enumerate(segments):
        if si in round_trip_skip:
            continue
        dx, dy, tx, ty = layout.place_dist_time(
            seg["distance"], seg["time"], si, segments, cities,
        )
        render_dist_time(ax, seg["distance"], seg["time"], dx, dy, tx, ty)

    # ── L6 景点 ──
    print("景点...")
    for idx, c in enumerate(cities):
        prim_list = c.get("attractions_primary", [])
        sec_list = c.get("attractions_secondary", [])

        # 城圈内景点：合并引线
        in_prim = [a for a in prim_list
                   if math.hypot(a["lon"] - c["lon"], a["lat"] - c["lat"])
                   < _city_radius(c["name"]) + 0.02]
        in_sec = [a for a in sec_list
                  if math.hypot(a["lon"] - c["lon"], a["lat"] - c["lat"])
                  < _city_radius(c["name"]) + 0.02]
        out_prim = [a for a in prim_list if a not in in_prim]
        out_sec = [a for a in sec_list if a not in in_sec]

        # 城圈内：合并
        if in_prim or in_sec:
            side = -1 if idx % 2 == 0 else 1
            result = layout.place_grouped_attractions(
                c["name"], c["lat"], c["lon"], idx,
                [a["name"] for a in in_prim],
                [a["name"] for a in in_sec],
                side,
            )
            if result:
                bx, by, side = result
                prim_str = "、".join(a["name"] for a in in_prim)
                sec_str = "、".join(a["name"] for a in in_sec)
                render_grouped_attractions(
                    ax, c["name"], c["lat"], c["lon"],
                    bx, by, side, prim_str, sec_str,
                )

        # 城圈外：单独打点
        for ai, a in enumerate(out_prim):
            tx, ty, ha, leader = layout.place_attraction(
                a["name"], a["lat"], a["lon"],
                c["name"], c["lat"], c["lon"],
                True, idx, ai,
            )
            render_attraction_outside(
                ax, a["name"], a["lat"], a["lon"],
                True, tx, ty, ha, leader,
            )
        for ai, a in enumerate(out_sec):
            tx, ty, ha, leader = layout.place_attraction(
                a["name"], a["lat"], a["lon"],
                c["name"], c["lat"], c["lon"],
                False, idx, ai + len(out_prim),
            )
            render_attraction_outside(
                ax, a["name"], a["lat"], a["lon"],
                False, tx, ty, ha, leader,
            )

    # ── 行程表 ──
    print("行程表...")
    table_lines = _build_itinerary_lines(cfg, all_days, day_to_segs)
    result = layout.place_itinerary(table_lines, extent, cities)
    if result:
        cx, cy, tw, th = result
        render_itinerary(ax, table_lines, cx, cy, tw, th)
        print(f"  单列: ({cx:.1f}, {cy:.1f})")
    else:
        # 尝试双列
        result2 = layout.place_itinerary_two_col(table_lines, extent, cities)
        if result2:
            mid = len(table_lines) // 2
            for (cx, cy, tw, th), col_lines in zip(
                result2, [table_lines[:mid], table_lines[mid:]]
            ):
                render_itinerary(ax, col_lines, cx, cy, tw, th)
            print("  双列")
        else:
            print("  行程表：未找到位置")

    # ── 标题 ──
    print("标题...")
    center_x = (extent[0] + extent[1]) / 2
    render_title(ax, cfg, center_x)

    # ── 输出 ──
    print("保存...")
    route_name = f"{cities[0]['name']}_{cities[-1]['name']}_{cfg['title']}"
    cache_dir = os.path.expanduser(output["cache_dir"])
    os.makedirs(cache_dir, exist_ok=True)
    out_path = os.path.join(cache_dir, f"{route_name}.png")

    plt.savefig(out_path, dpi=output["dpi"], facecolor="white")
    plt.close()

    if output.get("desktop_copy", False):
        desktop = os.path.expanduser(f"~/Desktop/{route_name}.png")
        shutil.copy2(out_path, desktop)
        print(f"  桌面副本: {desktop}")

    mb = os.path.getsize(out_path) / 1024 / 1024
    print(f"完成! {mb:.1f} MB → {out_path}")
    return out_path


def _build_itinerary_lines(cfg: dict, all_days: list,
                           day_to_segs: dict) -> list:
    """构建行程表文本行。"""
    cities = cfg["cities"]
    segments = cfg["segments"]
    rest_days = cfg.get("rest_days", {})
    day_attractions = cfg.get("day_attractions", {})
    all_seg_days = set(day_to_segs.keys())
    all_rest_days = set(int(k) for k in rest_days.keys())

    lines = []
    for day in all_days:
        seg_indices = day_to_segs.get(day, [])
        is_rest = str(day) in rest_days
        rc = rest_days.get(str(day))

        if is_rest and rc:
            if str(day) in day_attractions:
                da = day_attractions[str(day)]
                aa = da.get("primary", []) + da.get("secondary", [])
            else:
                ci = next(i for i, c in enumerate(cities) if c["name"] == rc)
                aa = [a["name"] for a in cities[ci].get("attractions_primary", [])] + \
                     [a["name"] for a in cities[ci].get("attractions_secondary", [])]
            line = f"D{day}  {rc}全天"
            if aa:
                line += f"  ·  {'、'.join(aa)}"
            lines.append(line)
        elif seg_indices:
            cr = []
            aa = []
            for si in seg_indices:
                s = segments[si]
                if not cr:
                    cr.append(cities[s["from_index"]]["name"])
                cr.append(cities[s["to_index"]]["name"])
                ci = s["to_index"]
                if str(day) in day_attractions:
                    da = day_attractions[str(day)]
                    aa.extend(da.get("primary", []) + da.get("secondary", []))
                else:
                    aa.extend([a["name"] for a in cities[ci].get("attractions_primary", [])])
                    aa.extend([a["name"] for a in cities[ci].get("attractions_secondary", [])])
            line = f"D{day}  {'→'.join(dict.fromkeys(cr))}"
            unique_attrs = list(dict.fromkeys(aa))
            if unique_attrs:
                line += f"  ·  {'、'.join(unique_attrs)}"
            lines.append(line)
        else:
            # 驻留日：沿用前一旅行日到达城市
            prev = [d for d in sorted(all_seg_days | all_rest_days | {0}) if d < day]
            city = ""
            if prev:
                pd = prev[-1]
                if pd in day_to_segs:
                    city = cities[segments[day_to_segs[pd][-1]]["to_index"]]["name"]
                elif str(pd) in rest_days:
                    city = rest_days[str(pd)]
            if str(day) in day_attractions:
                da = day_attractions[str(day)]
                aa = da.get("primary", []) + da.get("secondary", [])
                line = f"D{day}  {city}"
                if aa:
                    line += f"  ·  {'、'.join(aa)}"
            else:
                line = f"D{day}  {city}"
            lines.append(line)

    return lines


def _city_radius(name: str) -> float:
    if len(name) <= 2:
        return 0.13
    if len(name) == 3:
        return 0.14
    return 0.15


def main():
    parser = argparse.ArgumentParser(description="自驾路线图生成器")
    parser.add_argument("config", help="JSON 配置文件路径")
    parser.add_argument("--output", "-o", default=None,
                        help="输出文件路径（覆盖配置文件中的设置）")
    args = parser.parse_args()

    cfg = load_config(args.config)
    out = generate(cfg)

    if args.output:
        shutil.copy2(out, os.path.expanduser(args.output))
        print(f"→ {args.output}")

    return 0


if __name__ == "__main__":
    sys.exit(main())
```
