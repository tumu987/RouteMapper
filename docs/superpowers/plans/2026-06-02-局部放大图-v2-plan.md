# 局部放大镜 v2 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 为路线图生成引擎添加圆形放大镜——当相邻城市距离过近时，自动生成包含完整主图信息的 8 倍圆形放大镜。

**Architecture:** 独立 figure 渲染放大图（LayoutEngine 完整管道但跳过松弛）→ RGBA 截图 → 圆形裁剪 → `fig.figimage` 贴入主图。方图渲染范围是圆的 2 倍确保底图完整。主图跳过被放大城市的标注（天次/距离/景点）但保留城圈和路线。

**Tech Stack:** Python 3, Matplotlib + Cartopy + NumPy

```python
def _compute_zoom_extent(cluster_indices: set, cities: list,
                         padding: float) -> list:
    """计算放大镜的地理范围。

    Returns:
        [lon_min, lon_max, lat_min, lat_max, zoom_radius]
        zoom_radius 是圆形放大镜实际展示的地理半径。
        渲染范围 = zoom_radius * RENDER_MARGIN（方图比圆大一倍）。
    """
    from config import ZOOM_FACTOR, RENDER_MARGIN

    all_lons, all_lats = [], []
    for idx in cluster_indices:
        c = cities[idx]
        all_lons.append(c["lon"])
        all_lats.append(c["lat"])
        for attr_key in ("attractions_primary", "attractions_secondary"):
            for a in c.get(attr_key, []):
                try:
                    all_lons.append(float(a["lon"]))
                    all_lats.append(float(a["lat"]))
                except (ValueError, TypeError):
                    pass

    if not all_lons:
        return None

    center_lon = (min(all_lons) + max(all_lons)) / 2
    center_lat = (min(all_lats) + max(all_lats)) / 2
    max_dist = 0.0
    for lon, lat in zip(all_lons, all_lats):
        d = math.hypot(lon - center_lon, lat - center_lat)
        max_dist = max(max_dist, d)

    # zoom 半径：簇范围 × padding × (5/8) 基准缩放
    base_radius = max(max_dist * padding, 0.15)
    zoom_radius = base_radius * 5 / ZOOM_FACTOR
    zoom_radius = max(zoom_radius, max_dist * 1.3)

    # 渲染半径 = zoom 半径 × RENDER_MARGIN
    render_radius = zoom_radius * RENDER_MARGIN

    return [
        round(center_lon - render_radius, 3),
        round(center_lon + render_radius, 3),
        round(center_lat - render_radius, 3),
        round(center_lat + render_radius, 3),
        zoom_radius,
    ]
```

```python
def _place_zoom_inset(layout: LayoutEngine,
                      extent_zoom: list,
                      main_extent: list,
                      crs_cos: float,
                      inset_fig_w: float = 0.32) -> tuple:
    """螺旋搜索空白区域放置放大镜。避开行程表。
    
    Returns:
        (center_lon, center_lat, half_w, half_h)
    回退: 缩小 20% 重搜一轮；仍失败返回 None 标记放弃。
    """
    for attempt, scale in enumerate([1.0, 0.8]):
        inset_hw_map = (inset_fig_w * scale) * (main_extent[1] - main_extent[0]) / (2 * 0.98)
        inset_hh_map = (inset_fig_w * scale) * (main_extent[3] - main_extent[2]) / (2 * 0.98)

        center_lon = (extent_zoom[0] + extent_zoom[1]) / 2
        center_lat = (extent_zoom[2] + extent_zoom[3]) / 2

        for step in range(1, 20):
            radius = 1.5 + step * 0.3
            for ang_idx in range(8):
                angle = math.radians(ang_idx * 45 + step * 15)
                cx = center_lon + radius * math.cos(angle)
                cy = center_lat + radius * math.sin(angle)

                if not (main_extent[0] + inset_hw_map < cx < main_extent[1] - inset_hw_map):
                    continue
                if not (main_extent[2] + inset_hh_map < cy < main_extent[3] - inset_hh_map):
                    continue

                if layout.is_position_clear(cx, cy, inset_hw_map, inset_hh_map,
                                            margin=0.05, my_kind="inset"):
                    layout.place(cx, cy, inset_hw_map, inset_hh_map,
                                f"inset_{center_lon:.2f}_{center_lat:.2f}")
                    return cx, cy, inset_hw_map, inset_hh_map

    return None, None, 0, 0  # 放弃
```

def _render_zoom_pipeline(ax, extent_zoom: list, zoom_radius: float,
                          cluster: set, cities: list, segments: list,
                          day_colors: dict, output: dict) -> None:
    """在给定 axes 上运行完整布局管道（无松弛）。

    只处理簇内城市。外部连线只画线不参与布局。
    """
    from config import ZOOM_FACTOR
    cluster_list = sorted(cluster)
    local_cities = [cities[i] for i in cluster_list]
    idx_map = {old: new for new, old in enumerate(cluster_list)}

    local_segs = []
    external_segs = []
    for seg in segments:
        fi, ti = seg["from_index"], seg["to_index"]
        f_in, t_in = fi in cluster, ti in cluster
        if f_in and t_in:
            ls = dict(seg)
            ls["from_index"] = idx_map[fi]
            ls["to_index"] = idx_map[ti]
            local_segs.append(ls)
        elif f_in and not t_in:
            external_segs.append((fi, ti, seg))
        elif t_in and not f_in:
            external_segs.append((ti, fi, seg))

    local_day_colors = {}
    for seg in local_segs + [s for _, _, s in external_segs]:
        d = str(seg["day"])
        if d not in local_day_colors:
            local_day_colors[d] = day_colors.get(d, "#888888")

    avg_lat = sum(c["lat"] for c in local_cities) / len(local_cities)
    crs_cos = math.cos(math.radians(avg_lat))
    zoom_city_radius = CITY_RADIUS / ZOOM_FACTOR
    extent_w = extent_zoom[1] - extent_zoom[0]
    px_per_deg = (output.get("width_inch", 5.0) * 0.98 * output["dpi"]) / extent_w if extent_w > 0 else 200
    layout = LayoutEngine(px_per_deg, output["dpi"])

    # 外部连线
    from renderer import render_route_segment as _draw_route
    for fi, ti, seg in external_segs:
        _draw_route(ax, cities[fi]["lon"], cities[fi]["lat"],
                    cities[ti]["lon"], cities[ti]["lat"],
                    day_colors.get(str(seg["day"]), "#888888"))

    # 簇内路线
    from renderer import render_all_routes as _render_routes, render_city_node as _render_node
    route_pairs = _render_routes(ax, local_segs, local_cities, local_day_colors)
    for fi, ti in route_pairs:
        layout.register_route(local_cities[fi]["lon"], local_cities[fi]["lat"],
                              local_cities[ti]["lon"], local_cities[ti]["lat"],
                              LayoutEngine.ROUTE_HW)

    # 城市
    for c in local_cities:
        _render_node(ax, c["name"], c["lon"], c["lat"],
                     c.get("color", "#888888"), crs_cos,
                     radius=zoom_city_radius)
        layout.place(c["lon"], c["lat"],
                     zoom_city_radius + 0.02,
                     zoom_city_radius * crs_cos + 0.02,
                     f"city_{c['name']}")

    local_cfg = {"cities": local_cities, "segments": local_segs,
                 "day_colors": local_day_colors, "rest_days": {},
                 "day_attractions": {}, "output": output}

    day_to_segs = {}
    for si, seg in enumerate(local_segs):
        day_to_segs.setdefault(seg["day"], []).append(si)
    all_days = sorted(day_to_segs.keys())

    if not all_days:
        return

    day_recipes = _place_day_labels(layout, local_cfg, all_days, day_to_segs,
                                     local_cities, crs_cos)
    dt_recipes = _place_dist_time_labels(layout, local_segs, local_cities)
    attr_recipes = _place_attractions(layout, local_cities,
                                       city_radius=zoom_city_radius)

    # 不松弛！方图范围大一倍，标注天然分散
    # layout.relax_overlaps()

    _render_day_items(ax, layout, day_recipes)
    _render_attr_items(ax, layout, attr_recipes)
    _render_dt_items(ax, layout, dt_recipes)
```

def render_zoom_indicator(ax_main: Any, extent_zoom: list,
                          zoom_radius: float, crs_cos: float,
                          color: str = "#E74C3C") -> None:
    """在主图上绘制虚线圆选取框（与放大镜展示范围一致）。

    Args:
        ax_main: 主图 axes
        extent_zoom: 渲染范围 [lon_min, lon_max, lat_min, lat_max]
        zoom_radius: 选取框地理半径（与放大镜一致）
        crs_cos: 纬度补偿
        color: 颜色
    """
    center_lon = (extent_zoom[0] + extent_zoom[1]) / 2
    center_lat = (extent_zoom[2] + extent_zoom[3]) / 2
    rx = zoom_radius
    ry = zoom_radius * crs_cos  # 补偿纬度，显示为正圆

    n_pts = 64
    angles = [2 * math.pi * i / n_pts for i in range(n_pts + 1)]
    lons = [center_lon + rx * math.cos(a) for a in angles]
    lats = [center_lat + ry * math.sin(a) for a in angles]
    ax_main.plot(lons, lats, color=color, linewidth=1.5,
                 linestyle="--", transform=ccrs.PlateCarree(),
                 zorder=98, alpha=0.7)
```
